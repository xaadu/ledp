Here is a function to get Corona Information of Bangladesh from Worldometers: 

import requests
from bs4 import BeautifulSoup

def getCovidText():
	link = 'https://www.worldometers.info/coronavirus/country/bangladesh/'
	data=BeautifulSoup(requests.get(link).text, 'html.parser')
	src=data.find_all('a', {'class': 'news_source_a'})[0]['href']
	try:
		caseList = data.findAll('div', {'id': 'maincounter-wrap'})
		casesNumbers=[]
		for case in caseList:
			casesNumbers.append(case.div.span.text)
		text = "COVID-19 Stats: Bangladesh\n\nTotal Cases: {}\n\nTotal Recovered: {}\n\nTotal Death: {}\n\n"
		text=text.format(casesNumbers[0], casesNumbers[2], casesNumbers[1])
		try:
			news = data.find('ul', {'class': 'news_ul'})
			news=' '.join(news.text.split()[:-3])
		except Exception as e:
			print(e)
			news=None
		if news is not None: text+=news+'\n\n'
		return text+'Source: Worldometers'
	except Exception as e:
		return None

print(getCovidText())
